扩展性 难以添加高级功能 RK3588支持深度学习、语音、视觉

5.3 关节控制器

// joint_controller.c
#include "joint_controller.h"
#include <math.h>

#define NUM_JOINTS 12

static Joint_t joints[NUM_JOINTS];

void JointController_Init(void) {
    for (int i = 0; i < NUM_JOINTS; i++) {
        joints[i].id = i + 1;
        joints[i].position = 0.0f;
        joints[i].velocity = 0.0f;
        joints[i].torque = 0.0f;
        joints[i].target_pos = 0.0f;
        joints[i].target_vel = 0.0f;
        joints[i].kp = 10.0f;
        joints[i].kd = 1.0f;
        joints[i].enabled = 0;
    }
}

void JointController_Update(void) {
    // 1kHz控制频率
    for (int i = 0; i < NUM_JOINTS; i++) {
        if (!joints[i].enabled) continue;
        
        // PD控制计算
        float pos_error = joints[i].target_pos - joints[i].position;
        float vel_error = joints[i].target_vel - joints[i].velocity;
        
        float torque_cmd = joints[i].kp * pos_error + 
                          joints[i].kd * vel_error + 
                          joints[i].target_torque;
        
        // 力矩限幅
        if (torque_cmd > 20.0f) torque_cmd = 20.0f;
        if (torque_cmd < -20.0f) torque_cmd = -20.0f;
        
        joints[i].torque = torque_cmd;
    }
}

5.4 步态生成器

// gait_generator.c

// Trot步态足端轨迹生成
void generateTrotGait(double time, double period,
                      double step_length, double step_height,
                      std::vector<double>& foot_heights) {
    // Trot步态：对角腿同步
    double phase0 = fmod(time / period, 1.0);
    double phase1 = fmod(phase0 + 0.5, 1.0);
    
    foot_heights.resize(4);
    foot_heights[0] = cycloidZ(phase0, step_height);
    foot_heights[1] = cycloidZ(phase1, step_height);
    foot_heights[2] = cycloidZ(phase1, step_height);
    foot_heights[3] = cycloidZ(phase0, step_height);
}

// 摆线轨迹
double cycloidZ(double phase, double height) {
    if (phase < 0.5) {
        // 摆动相
        double theta = phase * 4.0 * M_PI;
        return height * (1.0 - cos(theta)) / 2.0;
    }
    return 0.0;  // 支撑相
}

// can_interface.cpp
#include <rclcpp/rclcpp.hpp>
#include <sensor_msgs/msg/joint_state.hpp>
#include <linux/can.h>
#include <sys/socket.h>

namespace quadruped_hardware {

class CANInterface : public rclcpp::Node {
public:
    CANInterface() : Node("can_interface") {
        this->declare_parameter("can_interface", "can0");
        
        initCAN();
        
        joint_state_pub_ = this->create_publisher<sensor_msgs::msg::JointState>(
            "joint_states", 10);
        
        timer_ = this->create_wall_timer(
            std::chrono::milliseconds(10),
            std::bind(&CANInterface::controlLoop, this));
    }

private:
    void initCAN() {
        socket_ = socket(PF_CAN, SOCK_RAW, CAN_RAW);
        // ... CAN初始化代码
    }
    
    void controlLoop() {
        struct can_frame frame;
        sensor_msgs::msg::JointState joint_state;
        
        // 读取CAN数据
        while (read(socket_, &frame, sizeof(frame)) > 0) {
            if (frame.can_id >= 1 && frame.can_id <= 12) {
                size_t idx = frame.can_id - 1;
                int16_t pos = frame.data[0] | (frame.data[1] << 8);
                joint_state.position[idx] = pos / 100.0 * M_PI / 180.0;
            }
        }
        
        joint_state_pub_->publish(joint_state);
    }
    
    int socket_;
    rclcpp::Publisher<sensor_msgs::msg::JointState>::SharedPtr joint_state_pub_;
};

}  // namespace quadruped_hardware

身体坐标系 {B} │ │ L0 (髋部偏移) ▼ 髋关节 {H} ────────●──────── 髋关节旋转 (θ1, 绕X轴) │ │ L1 (大腿长度) │ ▼ 膝关节 {K} ─────────●─────── 膝关节旋转 (θ2, 绕Y轴) │ │ L2 (小腿长度) │ ▼ 踝关节 {A} ─────────●─────── 踝关节旋转 (θ3, 绕Y轴) │ │ L3 (足部长度) ▼ 足端 {F} ●

7.2 逆运动学

// 逆运动学：足端位置 → 关节角度
bool inverseKinematics(double x, double y, double z,
                       double L0, double L1, double L2, double L3,
                       double& theta1, double& theta2, double& theta3) {
    // 髋关节角度
    theta1 = atan2(y - L0, z);
    
    // 转换到腿部平面坐标系
    double d = sqrt(x*x + (y-L0)*(y-L0) + z*z);
    double z_leg = sqrt((y-L0)*(y-L0) + z*z);
    
    // 检查可达性
    double max_reach = L1 + L2 + L3;
    double min_reach = fabs(L1 - L2 - L3);
    if (d > max_reach || d < min_reach) {
        return false;  // 不可达
    }
    
    // 膝关节角度 (余弦定理)
    double cos_knee = (L1*L1 + L2*L2 - d*d) / (2*L1*L2);
    theta2 = M_PI - acos(cos_knee);
    
    // 踝关节角度
    double alpha = atan2(x, z_leg);
    double beta = acos((L1*L1 + d*d - L2*L2) / (2*L1*d));
    theta3 = alpha + beta;
    
    return true;
}

软件安全：添加看门狗，实现故障检测，设置最大速度/力矩限制
