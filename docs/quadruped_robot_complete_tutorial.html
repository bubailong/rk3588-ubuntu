2.2.3 电流采样电路计算

; 采样电阻: R_sense = 0.01Ω (10mΩ)
; 放大倍数: G = 50 (INA199A1)
; 基准电压: V_ref = 1.65V (半量程偏置)

; 电流计算公式:
I = (V_ref - V_ADC) / (G × R_sense)

; 示例:
; ADC读数: 1.65V → 电流 = 0A
; ADC读数: 1.15V → 电流 = (1.65-1.15)/(50×0.01) = 1A
; ADC读数: 2.15V → 电流 = (1.65-2.15)/(50×0.01) = -1A

; 最大测量电流:
I_max = 1.65V / (50 × 0.01Ω) = 3.3A (单向)
; 实际双向范围: -3.3A ~ +3.3A

4.2 FOC算法核心代码

4.2.1 Clarke变换 (三相静止坐标系 → 两相静止坐标系)

// foc.c - Clarke变换实现
#include "foc.h"
#include 

// Clarke变换: Iα = Ia, Iβ = (Ia + 2*Ib)/√3
void clarke_transform(float ia, float ib, float ic, 
                      float* ialpha, float* ibeta) {
    // 检查输入电流之和是否为零 (KCL定律)
    float sum = ia + ib + ic;
    if (fabs(sum) > 0.1f) {
        // 错误: 电流传感器校准错误或接线错误
        Error_Handler(ERROR_CURRENT_SENSOR);
    }
    
    *ialpha = ia;
    *ibeta = (ia + 2.0f * ib) / 1.73205080757f;  // √3 ≈ 1.732
}

// 错误处理函数
void Error_Handler(ErrorCode_t error) {
    // 关闭所有PWM输出
    HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_1);
    HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_2);
    HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_3);
    
    // 设置错误标志
    g_error_code = error;
    
    // 进入安全状态
    g_motor_state = MOTOR_STATE_ERROR;
    
    // 发送错误信息到上位机
    CAN_SendError(error);
}

4.2.2 Park变换 (静止坐标系 → 旋转坐标系)

// Park变换: Id = Iα*cos(θ) + Iβ*sin(θ), Iq = -Iα*sin(θ) + Iβ*cos(θ)
void park_transform(float ialpha, float ibeta, float theta,
                    float* id, float* iq) {
    float cos_theta = cosf(theta);
    float sin_theta = sinf(theta);
    
    *id = ialpha * cos_theta + ibeta * sin_theta;
    *iq = -ialpha * sin_theta + ibeta * cos_theta;
}

// 反Park变换: Vα = Vd*cos(θ) - Vq*sin(θ), Vβ = Vd*sin(θ) + Vq*cos(θ)
void inverse_park_transform(float vd, float vq, float theta,
                            float* valpha, float* vbeta) {
    float cos_theta = cosf(theta);
    float sin_theta = sinf(theta);
    
    *valpha = vd * cos_theta - vq * sin_theta;
    *vbeta = vd * sin_theta + vq * cos_theta;
}

4.2.3 SVPWM生成

// SVPWM空间矢量调制
void svpwm_generate(float valpha, float vbeta, float vdc,
                    uint16_t* pwm_u, uint16_t* pwm_v, uint16_t* pwm_w) {
    // 计算扇区
    float v1 = vbeta;
    float v2 = 0.86602540378f * valpha - 0.5f * vbeta;  // √3/2
    float v3 = -0.86602540378f * valpha - 0.5f * vbeta;
    
    uint8_t sector = 0;
    if (v1 > 0) sector |= 1;
    if (v2 > 0) sector |= 2;
    if (v3 > 0) sector |= 4;
    
    // 计算占空比 (简化版)
    float x = vbeta;
    float y = 0.86602540378f * valpha + 0.5f * vbeta;
    float z = -0.86602540378f * valpha + 0.5f * vbeta;
    
    float t1, t2, t0;
    
    switch (sector) {
        case 1:  // 扇区I
            t1 = z;
            t2 = y;
            break;
        case 2:  // 扇区II
            t1 = y;
            t2 = -x;
            break;
        // ... 其他扇区
        default:
            t1 = 0;
            t2 = 0;
            break;
    }
    
    // 限幅
    if (t1 + t2 > 1.0f) {
        float scale = 1.0f / (t1 + t2);
        t1 *= scale;
        t2 *= scale;
    }
    
    t0 = 1.0f - t1 - t2;  // 零矢量时间
    
    // 转换为PWM寄存器值
    uint16_t period = htim1.Instance->ARR;  // PWM周期值
    
    *pwm_u = (uint16_t)((0.5f - t1 - t2 * 0.5f) * period);
    *pwm_v = (uint16_t)((0.5f + t1 * 0.5f - t2 * 0.5f) * period);
    *pwm_w = (uint16_t)((0.5f + t1 * 0.5f + t2) * period);
}

#define ENCODER_CPR  16384  // 14位分辨率 (2^14)
#define SPI_TIMEOUT  10     // SPI超时时间(ms)

static float encoder_offset = 0.0f;  // 零点偏移
static int32_t encoder_turns = 0;     // 转数计数
static uint16_t last_raw = 0;

// 初始化编码器
void Encoder_Init(void) {
    // 拉低片选
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET);
    HAL_Delay(1);
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_SET);
    
    // 读取当前位置作为零点
    encoder_offset = Encoder_GetAngle();
    encoder_turns = 0;
    last_raw = 0;
}

// 读取编码器角度 (0~2π rad)
float Encoder_GetAngle(void) {
    uint8_t tx_data[2] = {0x00, 0x00};  // 发送空数据触发读取
    uint8_t rx_data[2] = {0};
    
    // 拉低片选
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET);
    
    // SPI传输
    HAL_StatusTypeDef status = HAL_SPI_TransmitReceive(&hspi1, tx_data, rx_data, 2, SPI_TIMEOUT);
    
    // 拉高片选
    HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_SET);
    
    if (status != HAL_OK) {
        // SPI通信错误
        Error_Handler(ERROR_ENCODER_SPI);
        return 0.0f;
    }
    
    // 解析数据 (14位角度值)
    uint16_t raw_angle = ((rx_data[0] << 8) | rx_data[1]) >> 2;  // 去掉低2位状态位
    
    // 检测过零
    int16_t diff = (int16_t)raw_angle - (int16_t)last_raw;
    if (diff > ENCODER_CPR / 2) {
        encoder_turns--;  // 反向过零
    } else if (diff < -ENCODER_CPR / 2) {
        encoder_turns++;  // 正向过零
    }
    last_raw = raw_angle;
    
    // 计算总角度
    float angle = ((float)raw_angle / ENCODER_CPR) * 2.0f * M_PI;
    angle += encoder_turns * 2.0f * M_PI;
    
    return angle - encoder_offset;
}

#include "can_protocol.h"

#define TX_FIFO_SIZE  32
#define RX_FIFO_SIZE  32

static FDCAN_TxHeaderTypeDef tx_header;
static uint8_t tx_data[8];

// 初始化CAN
void CAN_Init(void) {
    // 配置过滤器 - 接收所有标准帧
    FDCAN_FilterTypeDef sFilterConfig;
    sFilterConfig.IdType = FDCAN_STANDARD_ID;
    sFilterConfig.FilterIndex = 0;
    sFilterConfig.FilterType = FDCAN_FILTER_MASK;
    sFilterConfig.FilterConfig = FDCAN_FILTER_TO_RXFIFO0;
    sFilterConfig.FilterID1 = 0x000;  // 接收所有ID
    sFilterConfig.FilterID2 = 0x000;  // 不屏蔽任何位
    
    if (HAL_FDCAN_ConfigFilter(&hfdcan1, &sFilterConfig) != HAL_OK) {
        Error_Handler(ERROR_CAN_FILTER);
    }
    
    // 启动CAN
    if (HAL_FDCAN_Start(&hfdcan1) != HAL_OK) {
        Error_Handler(ERROR_CAN_START);
    }
    
    // 使能中断
    if (HAL_FDCAN_ActivateNotification(&hfdcan1, 
        FDCAN_IT_RX_FIFO0_NEW_MESSAGE, 0) != HAL_OK) {
        Error_Handler(ERROR_CAN_INTERRUPT);
    }
}

// 发送关节状态
void CAN_SendJointStatus(uint8_t joint_id, JointStatus_t* status) {
    tx_header.Identifier = joint_id;
    tx_header.IdType = FDCAN_STANDARD_ID;
    tx_header.TxFrameType = FDCAN_DATA_FRAME;
    tx_header.DataLength = FDCAN_DLC_BYTES_8;
    tx_header.ErrorStateIndicator = FDCAN_ESI_ACTIVE;
    tx_header.BitRateSwitch = FDCAN_BRS_ON;  // 使用数据段高速率
    tx_header.FDFormat = FDCAN_FD_CAN;
    tx_header.TxEventFifoControl = FDCAN_NO_TX_EVENTS;
    tx_header.MessageMarker = 0;
    
    // 打包数据
    tx_data[0] = status->position & 0xFF;
    tx_data[1] = (status->position >> 8) & 0xFF;
    tx_data[2] = status->velocity & 0xFF;
    tx_data[3] = (status->velocity >> 8) & 0xFF;
    tx_data[4] = status->torque & 0xFF;
    tx_data[5] = (status->torque >> 8) & 0xFF;
    tx_data[6] = status->temperature;
    tx_data[7] = status->error_code;
    
    // 添加到发送队列
    if (HAL_FDCAN_AddMessageToTxFifoQ(&hfdcan1, &tx_header, tx_data) != HAL_OK) {
        // 发送队列满
        g_can_tx_overflow++;
    }
}

// CAN接收中断回调
void HAL_FDCAN_RxFifo0Callback(FDCAN_HandleTypeDef *hfdcan, uint32_t RxFifo0ITs) {
    FDCAN_RxHeaderTypeDef rx_header;
    uint8_t rx_data[8];
    
    if ((RxFifo0ITs & FDCAN_IT_RX_FIFO0_NEW_MESSAGE) != RESET) {
        // 读取消息
        if (HAL_FDCAN_GetRxMessage(hfdcan, FDCAN_RX_FIFO0, &rx_header, rx_data) == HAL_OK) {
            // 解析命令
            CAN_ParseCommand(rx_header.Identifier, rx_data, rx_header.DataLength);
        }
    }
}

// 解析接收到的命令
void CAN_ParseCommand(uint32_t can_id, uint8_t* data, uint32_t len) {
    if (can_id == CAN_ID_COMMAND) {
        // 解析控制指令
        JointCommand_t cmd;
        cmd.position = data[0] | (data[1] << 8);
        cmd.velocity = data[2] | (data[3] << 8);
        cmd.torque = data[4] | (data[5] << 8);
        cmd.kp = data[6];
        cmd.kd = data[7];
        
        // 应用到电机控制
        Motor_SetCommand(&cmd);
    }
}

    HAL_TIM_Base_Start_IT(&htim1);
    
    // 主循环
    while (1) {
        // 1kHz任务
        if (g_control_count % 20 == 0) {
            // 更新状态机
            MotorControl_UpdateState();
            
            // 发送状态到上位机
            JointStatus_t status;
            Motor_GetStatus(&status);
            CAN_SendJointStatus(MOTOR_ID, &status);
        }
        
        // 100Hz任务
        if (g_control_count % 200 == 0) {
            // 温度监控
            if (Motor_GetTemperature() > 80) {
                Motor_SetState(MOTOR_STATE_OVERHEAT);
            }
        }
        
        HAL_Delay(1);
    }
}

        uint16_t pwm_u, pwm_v, pwm_w;
        svpwm_generate(vd, vq, VDC_VOLTAGE, &pwm_u, &pwm_v, &pwm_w);

        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, pwm_u);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_2, pwm_v);

            default_value=map_file,
            description='Full path to map yaml file'),
        
        # 启动SLAM (如果启用)
        IncludeLaunchDescription(
            PythonLaunchDescriptionSource(
                os.path.join(pkg_dir, 'launch', 'slam.launch.py')),
            condition=IfCondition(slam),
            launch_arguments={'use_sim_time': use_sim_time}.items()
        ),
        
        # 启动Nav2 Bringup
        IncludeLaunchDescription(
            PythonLaunchDescriptionSource(
                os.path.join(get_package_share_directory('nav2_bringup'),
                            'launch', 'bringup_launch.py')),
            launch_arguments={
                'use_sim_time': use_sim_time,
                'params_file': nav2_params,
                'map': map_file_arg,
                'slam': 'false'  # SLAM单独启动
            }.items()
        ),
        
        # 启动RViz2
        Node(
            package='rviz2',
            executable='rviz2',
            name='rviz2',
            arguments=['-d', os.path.join(pkg_dir, 'rviz', 'nav2.rviz')],
            parameters=[{'use_sim_time': use_sim_time}],
            output='screen'
        ),
    ])

6.2 软件故障排查

6.2.1 编译错误

# 错误1: undefined reference to `sinf'
# 原因: 未链接数学库
# 解决: 在CMakeLists.txt中添加
target_link_libraries(your_target m)

# 错误2: multiple definition of `g_variable'
# 原因: 全局变量在头文件中定义
# 解决: 头文件中使用extern声明，在.c文件中定义

# 错误3: implicit declaration of function
# 原因: 头文件未包含或函数名拼写错误
# 解决: 检查#include和函数名拼写

6.2.2 运行时错误

# 错误1: HardFault_Handler
# 原因: 空指针访问、数组越界、栈溢出
# 排查:
  1. 查看LR寄存器确定错误位置
  2. 检查最近修改的代码
  3. 增加栈大小(Stack_Size)

# 错误2: 控制周期不稳定
# 原因: 中断优先级错误、DMA冲突
# 排查:
  1. 用示波器测量PWM频率
  2. 检查中断优先级配置
  3. 禁用不必要的DMA请求

# 错误3: CAN发送失败
# 原因: 发送队列满、总线错误
# 排查:
  1. 检查g_can_tx_overflow计数
  2. 测量CAN总线波形
  3. 检查终端电阻
